REQUIREMENT 5
Dockerized Deployment
Production-ready containerized deployment with example datasets and MCP server mode
DOCKERFILE
# PII-Safe — Production Docker Image
FROM python:3.11-slim
 
WORKDIR /app
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt
 
COPY . .
 
# Non-root user for security
RUN adduser --disabled-password appuser
USER appuser
 
EXPOSE 8000
CMD ["uvicorn","app.main:app","--host","0.0.0.0","--port","8000"]
DOCKER COMPOSE
version: '3.8'
services:
  pii-safe:
    build: .
    ports: ["8000:8000"]
    volumes:
      - ./policies:/app/policies
      - ./data:/app/data
    environment:
      - DB_URL=sqlite:///audit.db
  redis:
    image: redis:alpine
    ports: ["6379:6379"]
